## Is Unique -

```cpp
#include<iostream>
using namespace std;

bool isUnique(string input);
int main(){

	string input = "ascdfdr";

	cout << isUnique(input);

	return 0;
}



bool isUnique(string input){
	if(input.length()>128 ) return false;

	bool res_arr[128] = {false};

	for (char &c:input){
		int index = (int) c;
		if(res_arr[index]) return false;
		res_arr[index] = true;
	}
	return true;
}
```
